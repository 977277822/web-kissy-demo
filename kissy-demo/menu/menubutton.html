button控件效果展示
属性 
        content：内容展示
        render：目标对象
        elCls ： 控件样式
        tooltip：鼠标悬停显示的文字
事件 
        //菜单项事件
       listeners : {
            click : function(ev){
                console.log(ev.target.get('content'));
            }
        }
button控件代码展示 
        /**
         * menu : 菜单模块
         * menubutton ： 按钮菜单模块
         */
        require(["menu", "menubutton", "menu/assets/dpl.css", "menubutton/assets/dpl.css"], function (Menu, MenuButton) {
            var menuButton = new MenuButton({
                content : "button",
                render : "#container",
                tooltip : "button",
                //下拉菜单是否和按钮宽度一致
                matchElWidth : false,
                elCls : "ks-button-info ks-button-large",
                menu : {
                    xclass : "popupmenu",
                    children : [
                        {
                            xclass : "menuitem",
                            content : "item1"
                        },
                        {
                            xclass : "menuitem",
                            content : "item2"
                        }
                    ],
                    listeners : {
                        click : function(ev){
                            console.log(ev.target.get('content'));
                        }
                    }
                }
            });
            menuButton.render();
        });
