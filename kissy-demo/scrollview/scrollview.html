scroll-view（滚动视图）控件效果展示
属性 
        content：内容展示
        render：目标对象
        width：控件宽度
        height：控件高度
事件
控件代码展示 
        require(['scroll-view',"scroll-view/assets/dpl.css"], function(ScrollView){
            var content = '';
            for (var i = 0; i < 10; i++) {
                content += '<p>This is ' + i + ' block</p>';
            }
            var myScrollView = new ScrollView({
                width : 318,
                height : 360,
                render : '#scroll-view-contaienr',
                content : content
            });
            myScrollView.render();
        })
